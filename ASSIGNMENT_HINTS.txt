// Suggested tests for the applicant to add as tests/CounterButton.test.ts
import { CounterButton } from '../src/components/CounterButton';

test('default starts at 0', () => {
  const b = new CounterButton();
  expect(b.getCount()).toBe(0);
});

test('constructor initial value', () => {
  const b = new CounterButton(5);
  expect(b.getCount()).toBe(5);
});

test('click increments and returns new value', () => {
  const b = new CounterButton();
  expect(b.click()).toBe(1);
  expect(b.click()).toBe(2);
});
